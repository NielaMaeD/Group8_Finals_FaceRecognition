**­ЮЌЦ­Юўѓ­ЮЌ╗­ЮЌ╗­ЮЌХ­ЮЌ╗­ЮЌ┤ ­ЮЌ╝­ЮЌ│ ­ЮЌЎ­ЮЌ«­ЮЌ░­ЮЌ▓ ­ЮЌЦ­ЮЌ▓­ЮЌ░­ЮЌ╝­ЮЌ┤­ЮЌ╗­ЮЌХ­ЮўЂ­ЮЌХ­ЮЌ╝­ЮЌ╗ ­ЮЌ╝­ЮЌ╗ ­ЮўЂ­ЮЌх­ЮЌ▓ ­ЮЌб­ЮЌ┐­ЮЌХ­ЮЌ┤­ЮЌХ­ЮЌ╗­ЮЌ«­ЮЌ╣ ­ЮЌќ­ЮЌ«­Юўђ­ЮўЂ ­ЮЌ╝­ЮЌ│ ­ЮЌћ­ЮўЃ­ЮЌ▓­ЮЌ╗­ЮЌ┤­ЮЌ▓­ЮЌ┐­Юўђ**

    file_name = "Avengers Original 6.jpg"
    unknown_image = face_recognition.load_image_file(file_name)
    unknown_image_to_draw = cv2.imread(file_name)
    
    face_locations = face_recognition.face_locations(unknown_image)
    face_encodings = face_recognition.face_encodings(unknown_image, face_locations)
    
    for (top,right, bottom, left), face_encoding in zip(face_locations, face_encodings):
      matches = face_recognition.compare_faces(known_face_encodings, face_encoding)
    
      name = "Unknown"
    
      face_distances = face_recognition.face_distance(known_face_encodings, face_encoding)
      best_match_index = np.argmin(face_distances)
      if matches[best_match_index]:
        name = known_face_names[best_match_index]
      cv2.rectangle(unknown_image_to_draw, (left, top), (right, bottom),(0,255,0),2)
      cv2.putText(unknown_image_to_draw,name, (left, bottom-20), cv2.FONT_HERSHEY_SIMPLEX,1.5,(0,0,255),2, cv2.LINE_AA)
    
    cv2_imshow(unknown_image_to_draw)

­ЮўЈ­Юўб­ЮўИ­Юўг­Юўд­Юў║­Юўд, ­Юў░­Юў│ ­Юўі­ЮўГ­Юўф­Юў»­Юўх ­ЮўЅ­Юўб­Юў│­Юўх­Юў░­Юў», ­Юўд­Юў«­ЮўБ­Юў░­ЮўЦ­Юўф­Юўд­Юў┤ ­Юўб ­ЮўХ­Юў»­Юўф­Юў▓­ЮўХ­Юўд ­Юўд­Юў╣­Юў▒­ЮўГ­Юў░­Юў│­Юўб­Юўх­Юўф­Юў░­Юў» ­Юў░­ЮўД ­ЮўЦ­ЮўХ­Юўб­ЮўГ ­Юўф­ЮўЦ­Юўд­Юў»­Юўх­Юўф­Юўх­Юўф­Юўд­Юў┤. ­Юўљ­Юў» ­ЮўЕ­Юўф­Юў┤ ­Юў│­Юўд­Юўб­ЮўГ ­Юўф­ЮўЦ­Юўд­Юў»­Юўх­Юўф­Юўх­Юў║, ­Юўі­ЮўГ­Юўф­Юў»­Юўх ­Юўф­Юў┤ ­Юўб­Юў» ­Юўд­Юў╣­Юўц­Юўд­Юў▒­Юўх­Юўф­Юў░­Юў»­Юўб­ЮўГ­ЮўГ­Юў║ ­Юў┤­Юўг­Юўф­ЮўГ­ЮўГ­Юўд­ЮўЦ ­Юўб­Юў│­Юўц­ЮўЕ­Юўд­Юў│ ­Юўб­Юў»­ЮўЦ ­Юў«­Юўб­Юў│­Юўг­Юў┤­Юў«­Юўб­Юў», ­Юўе­Юў│­Юў░­ЮўХ­Юў»­ЮўЦ­Юўф­Юў»­Юўе ­Юўх­ЮўЕ­Юўд ­Юўѕ­Юўи­Юўд­Юў»­Юўе­Юўд­Юў│­Юў┤ ­ЮўИ­Юўф­Юўх­ЮўЕ ­ЮўЕ­Юўф­Юў┤ ­ЮўЕ­ЮўХ­Юў«­Юўб­Юў» ­Юў▒­Юў│­Юўд­Юўц­Юўф­Юў┤­Юўф­Юў░­Юў» ­Юўб­Юў«­Юўф­ЮўЦ ­Юў┤­ЮўХ­Юў▒­Юўд­Юў│­ЮўЕ­ЮўХ­Юў«­Юўб­Юў» ­Юўб­ЮўБ­Юўф­ЮўГ­Юўф­Юўх­Юўф­Юўд­Юў┤. ­Юўѕ­Юў┤ ­ЮўЈ­Юўб­ЮўИ­Юўг­Юўд­Юў║­Юўд, ­ЮўЕ­Юўд ­ЮўБ­Юўд­Юўц­Юў░­Юў«­Юўд­Юў┤ ­Юўх­ЮўЕ­Юўд ­ЮўХ­Юў»­Юў┤­ЮўХ­Юў»­Юўе ­ЮўЕ­Юўд­Юў│­Юў░ ­ЮўИ­Юўф­Юўд­ЮўГ­ЮўЦ­Юўф­Юў»­Юўе ­Юўб ­ЮўБ­Юў░­ЮўИ ­Юўб­Юў»­ЮўЦ ­Юўб­Юў│­Юў│­Юў░­ЮўИ, ­Юў┤­ЮўЕ­Юў░­ЮўИ­Юўц­Юўб­Юў┤­Юўф­Юў»­Юўе ­ЮўХ­Юў»­Юў▒­Юўб­Юў│­Юўб­ЮўГ­ЮўГ­Юўд­ЮўГ­Юўд­ЮўЦ ­Юўб­Юўц­Юўц­ЮўХ­Юў│­Юўб­Юўц­Юў║ ­Юўб­Юў»­ЮўЦ ­Юўб­ЮўЦ­Юўб­Юў▒­Юўх­Юўб­ЮўБ­Юўф­ЮўГ­Юўф­Юўх­Юў║. ­Юўі­ЮўГ­Юўф­Юў»­Юўх'­Юў┤ ­Юўф­ЮўЦ­Юўд­Юў»­Юўх­Юўф­Юўх­Юў║ ­Юўб­Юў┤ ­Юўб ­ЮўД­Юўб­Юў«­Юўф­ЮўГ­Юў║ ­Юў«­Юўб­Юў» ­ЮўД­ЮўХ­Юў│­Юўх­ЮўЕ­Юўд­Юў│ ­Юўб­ЮўЦ­ЮўЦ­Юў┤ ­ЮўГ­Юўб­Юў║­Юўд­Юў│­Юў┤ ­Юўх­Юў░ ­ЮўЕ­Юўф­Юў┤ ­Юўц­ЮўЕ­Юўб­Юў│­Юўб­Юўц­Юўх­Юўд­Юў│, ­Юўд­Юў«­Юў▒­ЮўЕ­Юўб­Юў┤­Юўф­Юў╗­Юўф­Юў»­Юўе ­Юўх­ЮўЕ­Юўд ­Юў│­Юўд­ЮўГ­Юўб­Юўх­Юўб­ЮўБ­ЮўГ­Юўд ­Юўб­Юў┤­Юў▒­Юўд­Юўц­Юўх­Юў┤ ­Юў░­ЮўД ­ЮўЕ­Юўф­Юў┤ ­ЮўГ­Юўф­ЮўД­Юўд ­ЮўБ­Юўд­Юў║­Юў░­Юў»­ЮўЦ ­Юў┤­ЮўХ­Юў▒­Юўд­Юў│­ЮўЕ­Юўд­Юў│­Юў░ ­ЮўЦ­ЮўХ­Юўх­Юўф­Юўд­Юў┤. ­ЮўЏ­ЮўЕ­Юўд ­Юў┤­Юўф­Юў«­Юў▒­ЮўГ­Юўф­Юўц­Юўф­Юўх­Юў║ ­Юў░­ЮўД ­Юўб ­ЮўБ­Юў░­ЮўИ ­Юўц­Юў░­Юў»­Юўх­Юў│­Юўб­Юў┤­Юўх­Юў┤ ­ЮўИ­Юўф­Юўх­ЮўЕ ­Юўх­ЮўЕ­Юўд ­Юўе­Юў│­Юўб­Юў»­ЮўЦ­Юўд­ЮўХ­Юў│ ­Юў░­ЮўД ­Юў┤­ЮўХ­Юў▒­Юўд­Юў│­Юў▒­Юў░­ЮўИ­Юўд­Юў│­Юў┤, ­Юў║­Юўд­Юўх ­ЮўЈ­Юўб­ЮўИ­Юўг­Юўд­Юў║­Юўд'­Юў┤ ­Юў│­Юўд­Юў┤­Юўф­ЮўГ­Юўф­Юўд­Юў»­Юўц­Юўд ­Юўб­Юў»­ЮўЦ ­Юў┤­Юўх­Юў│­Юўб­Юўх­Юўд­Юўе­Юўф­Юўц ­ЮўБ­Юў│­Юўф­ЮўГ­ЮўГ­Юўф­Юўб­Юў»­Юўц­Юўд ­Юў«­Юўб­Юўг­Юўд ­ЮўЕ­Юўф­Юў« ­Юўб­Юў» ­Юўф­Юў»­ЮўЦ­Юўф­Юў┤­Юў▒­Юўд­Юў»­Юў┤­Юўб­ЮўБ­ЮўГ­Юўд ­Юўѕ­Юўи­Юўд­Юў»­Юўе­Юўд­Юў│. ­Юўі­ЮўГ­Юўф­Юў»­Юўх ­ЮўЅ­Юўб­Юў│­Юўх­Юў░­Юў»'­Юў┤ ­Юўц­ЮўЕ­Юўб­Юў│­Юўб­Юўц­Юўх­Юўд­Юў│ ­ЮўХ­Юў»­ЮўЦ­Юўд­Юў│­Юў┤­Юўц­Юў░­Юў│­Юўд­Юў┤ ­Юўх­ЮўЕ­Юўб­Юўх ­Юўд­Юўи­Юўд­Юў» ­Юўф­Юў» ­Юўб ­ЮўИ­Юў░­Юў│­ЮўГ­ЮўЦ ­Юў░­ЮўД ­Юўе­Юў░­ЮўЦ­Юў┤ ­Юўб­Юў»­ЮўЦ ­Юў«­Юў░­Юў»­Юў┤­Юўх­Юўд­Юў│­Юў┤, ­Юўб ­Юў«­Юў░­Юў│­Юўх­Юўб­ЮўГ ­ЮўИ­Юўф­Юўх­ЮўЕ ­ЮўХ­Юў»­ЮўИ­Юўб­Юўи­Юўд­Юў│­Юўф­Юў»­Юўе ­Юў┤­Юўг­Юўф­ЮўГ­ЮўГ ­Юўб­Юў»­ЮўЦ ­ЮўЦ­Юўд­Юўх­Юўд­Юў│­Юў«­Юўф­Юў»­Юўб­Юўх­Юўф­Юў░­Юў» ­Юўц­Юўб­Юў» ­Юўц­Юўб­Юў│­Юўи­Юўд ­Юўб ­Юў┤­Юўф­Юўе­Юў»­Юўф­ЮўД­Юўф­Юўц­Юўб­Юў»­Юўх ­Юў┤­Юў▒­Юўб­Юўц­Юўд ­ЮўИ­Юўф­Юўх­ЮўЕ­Юўф­Юў» ­Юўх­ЮўЕ­Юўд ­Юў┤­ЮўХ­Юў▒­Юўд­Юў│­ЮўЕ­Юўд­Юў│­Юў░ ­Юў»­Юўб­Юў│­Юў│­Юўб­Юўх­Юўф­Юўи­Юўд.

# **­ЮЉ╗­ЮЉ»­ЮЉг ­ЮЉ»­ЮЉ╝­ЮЉ│­ЮЉ▓**
  <div align="justify">
­ЮўЏ­ЮўЕ­Юўд ­ЮўЈ­ЮўХ­ЮўГ­Юўг, ­Юўб­ЮўГ­Юўх­Юўд­Юў│ ­Юўд­Юўе­Юў░ ­Юў░­ЮўД ­ЮўІ­Юў│. ­ЮўЅ­Юў│­ЮўХ­Юўц­Юўд ­ЮўЅ­Юўб­Юў»­Юў»­Юўд­Юў│, ­Юў│­Юўд­Юў▒­Юў│­Юўд­Юў┤­Юўд­Юў»­Юўх­Юў┤ ­Юўб ­ЮўХ­Юў»­Юўф­Юў▓­ЮўХ­Юўд ­ЮўЦ­ЮўХ­Юўб­ЮўГ­Юўф­Юўх­Юў║ ­Юўф­Юў» ­Юўх­ЮўЕ­Юўд ­Юў│­Юўд­Юўб­ЮўГ­Юў« ­Юў░­ЮўД ­Юў┤­ЮўХ­Юў▒­Юўд­Юў│­ЮўЕ­Юўд­Юў│­Юў░ ­Юўф­ЮўЦ­Юўд­Юў»­Юўх­Юўф­Юўх­Юўф­Юўд­Юў┤. ­Юўљ­Юў» ­ЮўЕ­Юўф­Юў┤ ­Юў│­Юўд­Юўб­ЮўГ ­Юўф­ЮўЦ­Юўд­Юў»­Юўх­Юўф­Юўх­Юў║, ­ЮўЅ­Юўб­Юў»­Юў»­Юўд­Юў│ ­Юўф­Юў┤ ­Юўб ­ЮўБ­Юў│­Юўф­ЮўГ­ЮўГ­Юўф­Юўб­Юў»­Юўх ­Юў┤­Юўц­Юўф­Юўд­Юў»­Юўх­Юўф­Юў┤­Юўх ­Юўе­Юў│­Юўб­Юў▒­Юў▒­ЮўГ­Юўф­Юў»­Юўе ­ЮўИ­Юўф­Юўх­ЮўЕ ­Юўх­ЮўЕ­Юўд ­Юўц­Юў░­Юў»­Юў┤­Юўд­Юў▓­ЮўХ­Юўд­Юў»­Юўц­Юўд­Юў┤ ­Юў░­ЮўД ­Юўе­Юўб­Юў«­Юў«­Юўб ­Юў│­Юўб­ЮўЦ­Юўф­Юўб­Юўх­Юўф­Юў░­Юў». ­Юўѕ­Юў┤ ­Юўх­ЮўЕ­Юўд ­ЮўЈ­ЮўХ­ЮўГ­Юўг, ­ЮўЕ­Юўд ­ЮўХ­Юў»­ЮўЦ­Юўд­Юў│­Юўе­Юў░­Юўд­Юў┤ ­Юўб ­Юў│­Юўб­ЮўЦ­Юўф­Юўц­Юўб­ЮўГ ­Юўх­Юў│­Юўб­Юў»­Юў┤­ЮўД­Юў░­Юў│­Юў«­Юўб­Юўх­Юўф­Юў░­Юў» ­Юўф­Юў»­Юўх­Юў░ ­Юўб
